36[[deprecated("Use structured logging with fields instead of format strings")]]
37inline std::string format_string(const char* fmt, ...) {
38 va_list args;
39 va_start(args, fmt);
40
41 // Get required buffer size
42 va_list args_copy;
43 va_copy(args_copy, args);
44 int size = vsnprintf(nullptr, 0, fmt, args_copy);
45 va_end(args_copy);
46
47 if (size < 0) {
48 va_end(args);
49 return fmt;
50 }
51
52 // Format the string
53 std::vector<char> buffer(size + 1);
54 vsnprintf(buffer.data(), buffer.size(), fmt, args);
55 va_end(args);
56
57 return std::string(buffer.data(), size);
58}
