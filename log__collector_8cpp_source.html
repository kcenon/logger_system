45class log_collector::impl {
46public:
47 explicit impl(std::size_t buffer_size)
48 : buffer_size_(buffer_size)
49 , running_(false) {
50 }
51
52 ~impl() {
53 stop();
54 }
55
56 bool enqueue(thread_module::log_level level,
57 const std::string& message,
58 const std::string& file,
59 int line,
60 const std::string& function,
61 const std::chrono::system_clock::time_point& timestamp) {
62 {
63 std::unique_lock<std::mutex> lock(queue_mutex_);
64
65 // Check if queue is full
66 if (queue_.size() >= buffer_size_) {
67 // Drop the message
68 return false;
69 }
70
71 // Create log_entry with optional source location
72 log_entry entry(level, message, timestamp);
73 if (!file.empty() || line != 0 || !function.empty()) {
74 entry.location = source_location{file, line, function};
75 }
76 queue_.push(std::move(entry));
77 }
78
79 queue_cv_.notify_one();
80 return true;
81 }
82
83 void add_writer(base_writer* writer) {
84 std::lock_guard<std::mutex> lock(writers_mutex_);
85 writers_.push_back(writer);
86 }
87
88 void clear_writers() {
89 std::lock_guard<std::mutex> lock(writers_mutex_);
90 writers_.clear();
91 }
92
93 void start() {
94 if (!running_.exchange(true)) {
95 worker_thread_ = std::thread(&impl::process_loop, this);
96 }
97 }
98
99 void stop() {
100 if (running_.exchange(false)) {
101 queue_cv_.notify_all();
102 if (worker_thread_.joinable()) {
103 worker_thread_.join();
104 }
105 }
106 }
107
108 void flush() {
109 // Process all remaining entries
110 std::unique_lock<std::mutex> lock(queue_mutex_);
111 while (!queue_.empty()) {
112 auto entry = std::move(queue_.front());
113 queue_.pop();
114 lock.unlock();
115
116 write_to_all(entry);
117
118 lock.lock();
119 }
120
121 // Flush all writers
122 std::lock_guard<std::mutex> writer_lock(writers_mutex_);
123 for (auto* writer : writers_) {
124 writer->flush();
125 }
126 }
127
128 std::pair<size_t, size_t> get_queue_metrics() const {
129 std::lock_guard<std::mutex> lock(queue_mutex_);
130 return {queue_.size(), buffer_size_};
131 }
132
133private:
134 void process_loop() {
135 while (running_.load()) {
136 std::unique_lock<std::mutex> lock(queue_mutex_);
137
138 // Wait for entries or shutdown
139 queue_cv_.wait(lock, [this] {
140 return !queue_.empty() || !running_.load();
141 });
142
143 // Process batch of entries
144 std::vector<log_entry> batch;
145 while (!queue_.empty() && batch.size() < 100) {
146 batch.push_back(std::move(queue_.front()));
147 queue_.pop();
148 }
149 lock.unlock();
150
151 // Write batch to all writers
152 for (const auto& entry : batch) {
153 write_to_all(entry);
154 }
155 }
156
157 // Process any remaining entries
158 flush();
159 }
160
161 void write_to_all(const log_entry& entry) {
162 std::lock_guard<std::mutex> lock(writers_mutex_);
163 for (auto* writer : writers_) {
164 // Ignore write failures for now
165 std::string file = entry.location ? entry.location->file.to_string() : "";
166 int line = entry.location ? entry.location->line : 0;
167 std::string function = entry.location ? entry.location->function.to_string() : "";
168
169 writer->write(entry.level, entry.message.to_string(), file,
170 line, function, entry.timestamp);
171 }
172 }
173
174private:
175 std::size_t buffer_size_;
176 std::atomic<bool> running_;
177 std::thread worker_thread_;
178
179 std::queue<log_entry> queue_;
180 mutable std::mutex queue_mutex_;
181 std::condition_variable queue_cv_;
182
183 std::vector<base_writer*> writers_;
184 std::mutex writers_mutex_;
185};
