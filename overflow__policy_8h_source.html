70struct overflow_stats {
71 std::atomic<size_t> total_messages{0};
72 std::atomic<size_t> dropped_messages{0};
73 std::atomic<size_t> blocked_count{0};
74 std::atomic<size_t> grow_count{0};
75 std::atomic<size_t> current_size{0};
76 std::atomic<size_t> max_size_reached{0};
77 std::chrono::milliseconds total_block_time{0};
78 std::chrono::milliseconds max_block_time{0};
79
80 // Default constructor
81 overflow_stats() = default;
82
83 // Copy constructor (needed because of atomic members)
84 overflow_stats(const overflow_stats& other)
85 : total_messages(other.total_messages.load()),
86 dropped_messages(other.dropped_messages.load()),
87 blocked_count(other.blocked_count.load()),
88 grow_count(other.grow_count.load()),
89 current_size(other.current_size.load()),
90 max_size_reached(other.max_size_reached.load()),
91 total_block_time(other.total_block_time),
92 max_block_time(other.max_block_time) {}
93
94 // Copy assignment operator
95 overflow_stats& operator=(const overflow_stats& other) {
96 if (this != &other) {
97 total_messages = other.total_messages.load();
98 dropped_messages = other.dropped_messages.load();
99 blocked_count = other.blocked_count.load();
100 grow_count = other.grow_count.load();
101 current_size = other.current_size.load();
102 max_size_reached = other.max_size_reached.load();
103 total_block_time = other.total_block_time;
104 max_block_time = other.max_block_time;
105 }
106 return *this;
107 }
108
109 double get_drop_rate() const {
110 size_t total = total_messages.load();
111 return total > 0 ?
112 static_cast<double>(dropped_messages.load()) / total * 100.0 : 0.0;
113 }
114
115 void reset() {
116 total_messages = 0;
117 dropped_messages = 0;
118 blocked_count = 0;
119 grow_count = 0;
120 current_size = 0;
121 max_size_reached = 0;
122 total_block_time = std::chrono::milliseconds{0};
123 max_block_time = std::chrono::milliseconds{0};
124 }
125};

401template<typename T>
402class overflow_queue {
403public:
404 explicit overflow_queue(size_t max_size,
405 std::unique_ptr<overflow_policy_base> policy = nullptr)
406 : max_size_(max_size)
407 , policy_(std::move(policy)) {
408 if (!policy_) {
409 policy_ = overflow_policy_factory::create(overflow_policy_type::drop_oldest);
410 }
411 }
412
418 bool push(T&& item) {
419 std::lock_guard<std::mutex> lock(mutex_);
420
421 if (queue_.size() >= max_size_) {
422 if (!policy_->handle_overflow(item, queue_, max_size_)) {
423 return false;
424 }
425 }
426
427 queue_.push(std::move(item));
428 cv_.notify_one();
429 return true;
430 }
431
438 bool pop(T& item, std::chrono::milliseconds timeout = std::chrono::milliseconds{100}) {
439 std::unique_lock<std::mutex> lock(mutex_);
440
441 if (!cv_.wait_for(lock, timeout, [this] { return !queue_.empty() || stopped_; })) {
442 return false;
443 }
444
445 if (stopped_) {
446 return false;
447 }
448
449 if (queue_.empty()) {
450 return false;
451 }
452
453 item = std::move(queue_.front());
454 queue_.pop();
455
456 // Notify block policy if space is available
457 if (auto* block = dynamic_cast<block_policy*>(policy_.get())) {
458 block->notify_space_available();
459 }
460
461 return true;
462 }
463
468 size_t size() const {
469 std::lock_guard<std::mutex> lock(mutex_);
470 return queue_.size();
471 }
472
477 bool empty() const {
478 std::lock_guard<std::mutex> lock(mutex_);
479 return queue_.empty();
480 }
481
485 void stop() {
486 std::lock_guard<std::mutex> lock(mutex_);
487 stopped_ = true;
488 cv_.notify_all();
489 }
490
495 void set_policy(std::unique_ptr<overflow_policy_base> policy) {
496 std::lock_guard<std::mutex> lock(mutex_);
497 policy_ = std::move(policy);
498 }
499
504 overflow_stats get_stats() const {
505 std::lock_guard<std::mutex> lock(mutex_);
506 return policy_->get_stats();
507 }
508
509private:
510 size_t max_size_;
511 std::queue<T> queue_;
512 std::unique_ptr<overflow_policy_base> policy_;
513 mutable std::mutex mutex_;
514 std::condition_variable cv_;
515 std::atomic<bool> stopped_{false};
516};
