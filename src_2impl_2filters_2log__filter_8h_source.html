167class composite_filter : public log_filter {
168public:
169 enum class logic_type {
170 AND, // All filters must pass
171 OR // At least one filter must pass
172 };
173
174 explicit composite_filter(logic_type logic = logic_type::AND)
175 : logic_(logic) {}
176
177 void add_filter(std::unique_ptr<log_filter> filter) {
178 filters_.push_back(std::move(filter));
179 }
180
181 bool should_log(log_level level,
182 const std::string& message,
183 const std::string& file,
184 int line,
185 const std::string& function) const override {
186 if (filters_.empty()) {
187 return true; // No filters = pass all
188 }
189
190 if (logic_ == logic_type::AND) {
191 for (const auto& filter : filters_) {
192 if (!filter->should_log(level, message, file, line, function)) {
193 return false;
194 }
195 }
196 return true;
197 } else { // OR logic
198 for (const auto& filter : filters_) {
199 if (filter->should_log(level, message, file, line, function)) {
200 return true;
201 }
202 }
203 return false;
204 }
205 }
206
207 std::string get_name() const override {
208 return "composite_filter";
209 }
210
211private:
212 logic_type logic_;
213 std::vector<std::unique_ptr<log_filter>> filters_;
214};
