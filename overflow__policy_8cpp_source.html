48bool drop_oldest_policy::handle_overflow(const log_entry& /* entry */,
49 std::queue<log_entry>& queue,
50 size_t /* max_size */) {
51 stats_.total_messages++;
52 stats_.dropped_messages++;
53
54 // Remove oldest entry
55 if (!queue.empty()) {
56 queue.pop();
57 }
58
59 // Return true to indicate the new entry can be added
60 return true;

74// Block policy implementation
75bool block_policy::handle_overflow(const log_entry& /* entry */,
76 std::queue<log_entry>& queue,
77 size_t max_size) {
78 stats_.total_messages++;
79 stats_.blocked_count++;
80
81 auto start_time = std::chrono::steady_clock::now();
82
83 std::unique_lock<std::mutex> lock(mutex_);
84 bool space_available = cv_.wait_for(lock, timeout_, [&queue, max_size] {
85 return queue.size() < max_size;
86 });
87
88 auto block_duration = std::chrono::duration_cast<std::chrono::milliseconds>(
89 std::chrono::steady_clock::now() - start_time);
90
91 stats_.total_block_time += block_duration;
92 if (block_duration > stats_.max_block_time) {
93 stats_.max_block_time = block_duration;
94 }
95
96 if (!space_available) {
97 stats_.dropped_messages++;
98 return false;
99 }
100
101 return true;
102}

104// Grow policy implementation
105bool grow_policy::handle_overflow(const log_entry& /* entry */,
106 std::queue<log_entry>& /* queue */,
107 size_t max_size) {
108 stats_.total_messages++;
109
110 size_t current_total = max_size + current_growth_.load();
111
112 if (current_total >= max_total_size_) {
113 // Cannot grow anymore, drop the message
114 stats_.dropped_messages++;
115 return false;
116 }
117
118 // Calculate new growth
119 size_t new_growth = std::min(
120 max_size * growth_factor_,
121 max_total_size_ - max_size
122 );
123
124 current_growth_ = new_growth;
125 stats_.grow_count++;
126
127 size_t new_max = max_size + new_growth;
128 if (new_max > stats_.max_size_reached) {
129 stats_.max_size_reached = new_max;
130 }
131
132 return true;
133}

135// Adaptive backpressure implementation
136class adaptive_backpressure::impl {
137public:
138 explicit impl(const config& cfg)
139 : config_(cfg)
140 , current_batch_size_(cfg.initial_batch_size)
141 , current_flush_interval_(cfg.initial_flush_interval)
142 , enabled_(true)
143 , adaptation_count_(0)
144 , increase_count_(0)
145 , decrease_count_(0) {
146 load_samples_.reserve(cfg.sample_count);
147 processing_times_.reserve(cfg.sample_count);
148 }
149
150 void update_metrics(double queue_usage, std::chrono::milliseconds processing_time) {
151 std::lock_guard<std::mutex> lock(mutex_);
152
153 // Add new samples
154 auto now = std::chrono::steady_clock::now();
155 load_samples_.push_back({queue_usage, now});
156 processing_times_.push_back({processing_time, now});
157
158 // Remove old samples outside the window
159 auto window_start = now - config_.metrics_window;
160
161 load_samples_.erase(
162 std::remove_if(load_samples_.begin(), load_samples_.end(),
163 [window_start](const auto& sample) {
164 return sample.timestamp < window_start;
165 }),
166 load_samples_.end()
167 );
168
169 processing_times_.erase(
170 std::remove_if(processing_times_.begin(), processing_times_.end(),
171 [window_start](const auto& sample) {
172 return sample.timestamp < window_start;
173 }),
174 processing_times_.end()
175 );
176
177 // Limit to sample count
178 if (load_samples_.size() > config_.sample_count) {
179 load_samples_.erase(load_samples_.begin(),
180 load_samples_.begin() + (load_samples_.size() - config_.sample_count));
181 }
182
183 if (processing_times_.size() > config_.sample_count) {
184 processing_times_.erase(processing_times_.begin(),
185 processing_times_.begin() + (processing_times_.size() - config_.sample_count));
186 }
187
188 // Calculate current load
189 if (!load_samples_.empty()) {
190 double sum = std::accumulate(load_samples_.begin(), load_samples_.end(), 0.0,
191 [](double acc, const auto& sample) {
192 return acc + sample.value;
193 });
194 current_load_ = sum / load_samples_.size();
195 }
196
197 // Trigger adaptation if enabled
198 if (enabled_) {
199 adapt_internal();
200 }
201 }
202
203 void adapt_internal() {
204 adaptation_count_++;
205
206 if (current_load_ > config_.load_threshold_high) {
207 // High load - increase batch size, decrease flush interval
208 increase_pressure();
209 increase_count_++;
210 } else if (current_load_ < config_.load_threshold_low) {
211 // Low load - decrease batch size, increase flush interval
212 decrease_pressure();
213 decrease_count_++;
214 }
215 }
216
217 void increase_pressure() {
218 // Increase batch size
219 size_t new_batch_size = static_cast<size_t>(
220 current_batch_size_ * (1.0 + config_.adaptation_rate)
221 );
222 current_batch_size_ = std::min(new_batch_size, config_.max_batch_size);
223
224 // Decrease flush interval
225 auto new_interval = std::chrono::milliseconds(static_cast<long>(
226 current_flush_interval_.count() * (1.0 - config_.adaptation_rate)
227 ));
228 current_flush_interval_ = std::max(new_interval, config_.min_flush_interval);
229 }
230
231 void decrease_pressure() {
232 // Decrease batch size
233 size_t new_batch_size = static_cast<size_t>(
234 current_batch_size_ * (1.0 - config_.adaptation_rate)
235 );
236 current_batch_size_ = std::max(new_batch_size, config_.min_batch_size);
237
238 // Increase flush interval
239 auto new_interval = std::chrono::milliseconds(static_cast<long>(
240 current_flush_interval_.count() * (1.0 + config_.adaptation_rate)
241 ));
242 current_flush_interval_ = std::min(new_interval, config_.max_flush_interval);
243 }
244
245 void reset() {
246 std::lock_guard<std::mutex> lock(mutex_);
247
248 current_batch_size_ = config_.initial_batch_size;
249 current_flush_interval_ = config_.initial_flush_interval;
250 current_load_ = 0.0;
251 adaptation_count_ = 0;
252 increase_count_ = 0;
253 decrease_count_ = 0;
254
255 load_samples_.clear();
256 processing_times_.clear();
257 }
258

260 std::lock_guard<std::mutex> lock(mutex_);
261 return current_batch_size_;
262 }
263

265 std::lock_guard<std::mutex> lock(mutex_);
266 return current_flush_interval_;
267 }

270 std::lock_guard<std::mutex> lock(mutex_);
271 return current_load_;
272 }
273
274 void set_enabled(bool enabled) {
275 std::lock_guard<std::mutex> lock(mutex_);
276 enabled_ = enabled;
277 }

280 std::lock_guard<std::mutex> lock(mutex_);
281 return enabled_;
282 }

285 std::lock_guard<std::mutex> lock(mutex_);
286

288 current_batch_size_,
289 current_flush_interval_,
290 current_load_,
291 adaptation_count_,
292 increase_count_,
293 decrease_count_
294 };
295 }

298 template<typename T>
299 struct timestamped_sample {
300 T value;
301 std::chrono::steady_clock::time_point timestamp;
302 };

305 mutable std::mutex mutex_;
306
307 // Current adaptive values
308 size_t current_batch_size_;
309 std::chrono::milliseconds current_flush_interval_;

311
312 // Metrics
313 std::vector<timestamped_sample<double>> load_samples_;
314 std::vector<timestamped_sample<std::chrono::milliseconds>> processing_times_;
315
316 // State
317 bool enabled_;

319 size_t increase_count_;
320 size_t decrease_count_;
321};

371// Custom policy implementation
372class custom_overflow_policy : public overflow_policy_base {
373public:
374 explicit custom_overflow_policy(
375 std::function<bool(const log_entry&, std::queue<log_entry>&, size_t)> handler)
376 : handler_(handler) {}
377
378 bool handle_overflow(const log_entry& entry,
379 std::queue<log_entry>& queue,
380 size_t max_size) override {
381 stats_.total_messages++;
382 bool result = handler_(entry, queue, max_size);
383 if (!result) {
384 stats_.dropped_messages++;
385 }
386 return result;
387 }
388
389 overflow_policy_type get_type() const override {
390 return overflow_policy_type::custom;
391 }
392
393private:
394 std::function<bool(const log_entry&, std::queue<log_entry>&, size_t)> handler_;
395};
