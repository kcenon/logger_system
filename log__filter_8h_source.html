173class composite_filter : public log_filter {
174public:
175 enum class logic_type {
176 AND, // All filters must pass
177 OR // At least one filter must pass
178 };
179
180 explicit composite_filter(logic_type logic = logic_type::AND)
181 : logic_(logic) {}
182
183 void add_filter(std::unique_ptr<log_filter> filter) {
184 filters_.push_back(std::move(filter));
185 }
186
187 bool should_log(thread_module::log_level level,
188 const std::string& message,
189 const std::string& file,
190 int line,
191 const std::string& function) const override {
192 if (filters_.empty()) {
193 return true; // No filters = pass all
194 }
195
196 if (logic_ == logic_type::AND) {
197 for (const auto& filter : filters_) {
198 if (!filter->should_log(level, message, file, line, function)) {
199 return false;
200 }
201 }
202 return true;
203 } else { // OR logic
204 for (const auto& filter : filters_) {
205 if (filter->should_log(level, message, file, line, function)) {
206 return true;
207 }
208 }
209 return false;
210 }
211 }
212
213 std::string get_name() const override {
214 return "composite_filter";
215 }
216
217private:
218 logic_type logic_;
219 std::vector<std::unique_ptr<log_filter>> filters_;
220};
