58 void apply(logger_config& config) const override {
59 switch (env_) {
60 case deployment_env::development:
61 apply_development(config);
62 break;
63 case deployment_env::staging:

66 case deployment_env::production:
67 apply_production(config);
68 break;
69 case deployment_env::testing:
70 apply_testing(config);

72 }
73 }

77 }
