103 common::VoidResult validate() const {
104 // Validate buffer size
105 if (buffer_size == 0) {
106 return make_logger_void_result(logger_error_code::invalid_configuration,
107 "Buffer size must be greater than 0");
108 }
109
110 if (buffer_size > std::numeric_limits<std::size_t>::max() / 2) {
111 return make_logger_void_result(logger_error_code::invalid_configuration,
112 "Buffer size is too large");
113 }
114
115 // Validate batch size
116 if (batch_size == 0) {
117 return make_logger_void_result(logger_error_code::invalid_configuration,
118 "Batch size must be greater than 0");
119 }
120
121 if (batch_size > buffer_size) {
122 return make_logger_void_result(logger_error_code::invalid_configuration,
123 "Batch size cannot exceed buffer size");
124 }
125
126 // Validate flush interval
127 if (flush_interval.count() < 0) {
128 return make_logger_void_result(logger_error_code::invalid_configuration,
129 "Flush interval must be non-negative");
130 }
131
132 if (flush_interval.count() > 3600000) { // 1 hour max
133 return make_logger_void_result(logger_error_code::invalid_configuration,
134 "Flush interval too large (max 1 hour)");
135 }
136
137 // Validate queue settings
138 if (max_queue_size == 0) {
139 return make_logger_void_result(logger_error_code::invalid_configuration,
140 "Max queue size must be greater than 0");
141 }
142
143 if (max_queue_size < batch_size) {
144 return make_logger_void_result(logger_error_code::invalid_configuration,
145 "Max queue size must be at least as large as batch size");
146 }
147
148 // Validate file settings
149 if (max_file_size < 1024) { // Minimum 1KB
150 return make_logger_void_result(logger_error_code::invalid_configuration,
151 "Max file size too small (minimum 1KB)");
152 }
153
154 if (max_file_count == 0) {
155 return make_logger_void_result(logger_error_code::invalid_configuration,
156 "Max file count must be greater than 0");
157 }
158
159 if (max_file_count > 1000) {
160 return make_logger_void_result(logger_error_code::invalid_configuration,
161 "Max file count too large (max 1000)");
162 }
163
164 // Validate network settings if configured
165 if (!remote_host.empty()) {
166 if (remote_port == 0) {
167 return make_logger_void_result(logger_error_code::invalid_configuration,
168 "Remote port must be specified when remote host is set");
169 }
170
171 if (network_timeout.count() <= 0) {
172 return make_logger_void_result(logger_error_code::invalid_configuration,
173 "Network timeout must be positive");
174 }
175
176 if (network_retry_count > 100) {
177 return make_logger_void_result(logger_error_code::invalid_configuration,
178 "Network retry count too large (max 100)");
179 }
180 }
181
182 // Validate writer settings
183 if (max_writers == 0) {
184 return make_logger_void_result(logger_error_code::invalid_configuration,
185 "Must allow at least one writer");
186 }
187
188 if (max_writers > 100) {
189 return make_logger_void_result(logger_error_code::invalid_configuration,
190 "Max writers too large (max 100)");
191 }
192
193 // Validate thread count
194 if (writer_thread_count == 0) {
195 return make_logger_void_result(logger_error_code::invalid_configuration,
196 "Writer thread count must be at least 1");
197 }
198
199 if (writer_thread_count > 32) {
200 return make_logger_void_result(logger_error_code::invalid_configuration,
201 "Writer thread count too large (max 32)");
202 }
203
204 // Validate feature combinations
205 if (use_lock_free && queue_overflow_policy == overflow_policy::grow) {
206 return make_logger_void_result(logger_error_code::invalid_configuration,
207 "Lock-free queue cannot use grow overflow policy");
208 }
209
210 if (!async && batch_size > 1) {
211 return make_logger_void_result(logger_error_code::invalid_configuration,
212 "Batch processing requires async mode");
213 }
214
215 return common::ok();
216 }
