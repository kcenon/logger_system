81 result_void validate() const {
82 // Validate buffer size
83 if (buffer_size == 0) {
84 return make_logger_error(logger_error_code::invalid_configuration,
85 "Buffer size must be greater than 0");
86 }
87
88 if (buffer_size > std::numeric_limits<std::size_t>::max() / 2) {
89 return make_logger_error(logger_error_code::invalid_configuration,
90 "Buffer size is too large");
91 }
92
93 // Validate batch size
94 if (batch_size == 0) {
95 return make_logger_error(logger_error_code::invalid_configuration,
96 "Batch size must be greater than 0");
97 }
98
99 if (batch_size > buffer_size) {
100 return make_logger_error(logger_error_code::invalid_configuration,
101 "Batch size cannot exceed buffer size");
102 }
103
104 // Validate flush interval
105 if (flush_interval.count() < 0) {
106 return make_logger_error(logger_error_code::invalid_configuration,
107 "Flush interval must be non-negative");
108 }
109
110 if (flush_interval.count() > 3600000) { // 1 hour max
111 return make_logger_error(logger_error_code::invalid_configuration,
112 "Flush interval too large (max 1 hour)");
113 }
114
115 // Validate queue settings
116 if (max_queue_size == 0) {
117 return make_logger_error(logger_error_code::invalid_configuration,
118 "Max queue size must be greater than 0");
119 }
120
121 if (max_queue_size < batch_size) {
122 return make_logger_error(logger_error_code::invalid_configuration,
123 "Max queue size must be at least as large as batch size");
124 }
125
126 // Validate file settings
127 if (max_file_size < 1024) { // Minimum 1KB
128 return make_logger_error(logger_error_code::invalid_configuration,
129 "Max file size too small (minimum 1KB)");
130 }
131
132 if (max_file_count == 0) {
133 return make_logger_error(logger_error_code::invalid_configuration,
134 "Max file count must be greater than 0");
135 }
136
137 if (max_file_count > 1000) {
138 return make_logger_error(logger_error_code::invalid_configuration,
139 "Max file count too large (max 1000)");
140 }
141
142 // Validate network settings if configured
143 if (!remote_host.empty()) {
144 if (remote_port == 0) {
145 return make_logger_error(logger_error_code::invalid_configuration,
146 "Remote port must be specified when remote host is set");
147 }
148
149 if (network_timeout.count() <= 0) {
150 return make_logger_error(logger_error_code::invalid_configuration,
151 "Network timeout must be positive");
152 }
153
154 if (network_retry_count > 100) {
155 return make_logger_error(logger_error_code::invalid_configuration,
156 "Network retry count too large (max 100)");
157 }
158 }
159
160 // Validate writer settings
161 if (max_writers == 0) {
162 return make_logger_error(logger_error_code::invalid_configuration,
163 "Must allow at least one writer");
164 }
165
166 if (max_writers > 100) {
167 return make_logger_error(logger_error_code::invalid_configuration,
168 "Max writers too large (max 100)");
169 }
170
171 // Validate thread count
172 if (writer_thread_count == 0) {
173 return make_logger_error(logger_error_code::invalid_configuration,
174 "Writer thread count must be at least 1");
175 }
176
177 if (writer_thread_count > 32) {
178 return make_logger_error(logger_error_code::invalid_configuration,
179 "Writer thread count too large (max 32)");
180 }
181
182 // Validate feature combinations
183 if (use_lock_free && queue_overflow_policy == overflow_policy::grow) {
184 return make_logger_error(logger_error_code::invalid_configuration,
185 "Lock-free queue cannot use grow overflow policy");
186 }
187
188 if (!async && batch_size > 1) {
189 return make_logger_error(logger_error_code::invalid_configuration,
190 "Batch processing requires async mode");
191 }
192
193 return result_void{};
194 }
