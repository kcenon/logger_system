312 result<std::unique_ptr<logger>> build() {
313 // Apply all strategies first
314 for (const auto& strategy : strategies_) {
315 if (auto can_apply = strategy->can_apply(config_); !can_apply) {
316 // Log warning but continue with other strategies
317 continue;
318 }
319
320 if (auto result = strategy->apply(config_); !result) {
321#ifdef USE_THREAD_SYSTEM
322 auto error_code = static_cast<int>(result.get_error().code());
323 auto logger_code = static_cast<logger_error_code>(error_code - 10000);
324 return make_error<std::unique_ptr<logger>>(
325 logger_code,
326 "Strategy application failed: " + result.get_error().message()
327 );
328#else
329 return make_logger_error<std::unique_ptr<logger>>(
330 result.error_code(),
331 "Strategy application failed: " + result.error_message()
332 );
333#endif
334 }
335 }
336
337 // Validate configuration
338 if (auto validation = config_.validate(); !validation) {
339#ifdef USE_THREAD_SYSTEM
340 // Extract error code - we need to reverse the offset we applied
341 auto error_code = static_cast<int>(validation.get_error().code());
342 auto logger_code = static_cast<logger_error_code>(error_code - 10000);
343 return make_error<std::unique_ptr<logger>>(
344 logger_code,
345 "Configuration validation failed: " + validation.get_error().message()
346 );
347#else
348 return make_logger_error<std::unique_ptr<logger>>(
349 validation.error_code(),
350 "Configuration validation failed: " + validation.error_message()
351 );
352#endif
353 }
354
355 // Validate writer count
356 if (!writers_.empty() && writers_.size() > config_.max_writers) {
357 return make_logger_error<std::unique_ptr<logger>>(
358 logger_error_code::invalid_configuration,
359 "Number of writers exceeds max_writers configuration"
360 );
361 }
362
363 // Create logger with validated configuration
364 auto logger_instance = std::make_unique<logger>(config_.async, config_.buffer_size);
365
366 // Apply configuration settings
367 logger_instance->set_min_level(config_.min_level);
368
369 if (config_.enable_metrics) {
370 logger_instance->enable_metrics_collection(true);
371 }
372
373 // Add writers
374 for (auto& [name, writer] : writers_) {
375 if (writer) {
376 writer->set_use_color(config_.enable_color_output);
377 logger_instance->add_writer(name, std::move(writer));
378 }
379 }
380
381 // Add filters
382 if (!filters_.empty()) {
383 // Create composite filter if multiple filters
384 if (filters_.size() == 1) {
385 logger_instance->set_filter(std::move(filters_[0]));
386 } else {
387 auto composite = std::make_unique<composite_filter>(composite_filter::logic_type::AND);
388 for (auto& filter : filters_) {
389 composite->add_filter(std::move(filter));
390 }
391 logger_instance->set_filter(std::move(composite));
392 }
393 }
394
395 // Start logger if async
396 if (config_.async) {
397 logger_instance->start();
398 }
399
400 // Store configuration in logger (if we add a config getter)
401 built_config_ = config_;
402
403 return std::move(logger_instance);
404 }
