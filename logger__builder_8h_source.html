241 result<std::unique_ptr<logger>> build() {
242 // Validate configuration
243 if (auto validation = config_.validate(); !validation) {
244 // Extract error code - we need to reverse the offset we applied
245 auto error_code = static_cast<int>(validation.get_error().code());
246 auto logger_code = static_cast<logger_error_code>(error_code - 10000);
247 return make_error<std::unique_ptr<logger>>(
248 logger_code,
249 "Configuration validation failed: " + validation.get_error().message()
250 );
251 }
252
253 // Validate writer count
254 if (!writers_.empty() && writers_.size() > config_.max_writers) {
255 return make_error<std::unique_ptr<logger>>(
256 logger_error_code::invalid_configuration,
257 "Number of writers exceeds max_writers configuration"
258 );
259 }
260
261 // Create logger with validated configuration
262 auto logger_instance = std::make_unique<logger>(config_.buffer_size);
263
264 // Apply configuration settings
265 logger_instance->set_min_level(config_.min_level);
266
267 if (config_.enable_metrics) {
268 logger_instance->enable_metrics_collection(true);
269 }
270
271 // Add writers
272 for (auto& [name, writer] : writers_) {
273 if (writer) {
274 writer->set_use_color(config_.enable_color_output);
275 logger_instance->add_writer(name, std::move(writer));
276 }
277 }
278
279 // Add filters
280 if (!filters_.empty()) {
281 // Create composite filter if multiple filters
282 if (filters_.size() == 1) {
283 logger_instance->set_filter(std::move(filters_[0]));
284 } else {
285 auto composite = std::make_unique<composite_filter>(composite_filter::logic_type::AND);
286 for (auto& filter : filters_) {
287 composite->add_filter(std::move(filter));
288 }
289 logger_instance->set_filter(std::move(composite));
290 }
291 }
292
293 // Start logger if async
294 if (config_.async) {
295 logger_instance->start();
296 }
297
298 // Store configuration in logger (if we add a config getter)
299 built_config_ = config_;
300
301 return std::move(logger_instance);
302 }
303
308 const logger_config& get_config() const {
309 return config_;
310 }

317 return config_.validate();
