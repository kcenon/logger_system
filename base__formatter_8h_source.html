124 oss << "\"thread\":\"" << (entry.thread_id ? *entry.thread_id : get_thread_id()) << "\"";
125
126 if (entry.location) {
127 oss << ",\"location\":{";
128 oss << "\"file\":\"" << escape_json(entry.location->file) << "\",";

130 oss << "\"function\":\"" << escape_json(entry.location->function) << "\"";
131 oss << "}";
132 }
133
134 if (entry.category) {

136 }

147private:
148 std::string escape_json(const std::string& str) const {
149 std::string escaped;
150 for (char c : str) {
151 if (c == '"') escaped += "\\\"";
152 else if (c == '\\') escaped += "\\\\";
153 else if (c == '\n') escaped += "\\n";
154 else if (c == '\r') escaped += "\\r";
155 else if (c == '\t') escaped += "\\t";
156 else if (c == '\b') escaped += "\\b";
157 else if (c == '\f') escaped += "\\f";
158 else escaped += c;
159 }
160 return escaped;
161 }
