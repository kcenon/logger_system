80 health_check_result perform_health_check() {
81 health_check_result result;
82 std::vector<health_status> all_statuses;
83
84 // Check writers
85 if (check_writers_) {
86 auto writer_statuses = check_all_writers();
87 for (const auto& [name, info] : writer_statuses) {
88 all_statuses.push_back(info.status);
89 if (info.status != health_status::healthy) {
90 result.add_issue("Writer '" + name + "': " +
91 format_writer_issues(info));
92 }
93 }
94 }
95
96 // Check buffers
97 if (check_buffers_) {
98 auto buffer_info = check_buffer_health_internal();
99 all_statuses.push_back(buffer_info.status);
100 if (buffer_info.status != health_status::healthy) {
101 result.add_issue("Buffer: " + format_buffer_issues(buffer_info));
102 }
103 }
104
105 // Check queues
106 if (check_queues_) {
107 auto queue_info = check_queue_health_internal();
108 all_statuses.push_back(queue_info.status);
109 if (queue_info.status != health_status::healthy) {
110 result.add_issue("Queue: " + format_queue_issues(queue_info));
111 }
112 }
113
114 // Run custom checks
115 for (const auto& [name, check] : custom_checks_) {
116 health_status status = check();
117 all_statuses.push_back(status);
118 if (status != health_status::healthy) {
119 result.add_issue("Custom check '" + name + "' failed");
120 }
121 }
122
123 // Determine overall health
124 if (!all_statuses.empty()) {
125 result.set_status(health_check_utils::aggregate_health_status(all_statuses));
126 }
127
128 if (result.get_status() == health_status::healthy) {
129 result.set_message("All systems healthy");
130 } else if (result.get_status() == health_status::degraded) {
131 result.set_message("System degraded - check issues");
132 } else {
133 result.set_message("System unhealthy - immediate attention required");
134 }
135
136 return result;
137 }

139 result<writer_health_info> check_writer_health(const std::string& name) {
140 std::lock_guard<std::mutex> lock(writers_mutex_);
141
142 auto it = writer_health_.find(name);
143 if (it == writer_health_.end()) {
144 return error_code::component_not_found;
145 }
146
147 return it->second;
148 }

150 buffer_health_info check_buffer_health_internal() {
151 std::lock_guard<std::mutex> lock(buffer_mutex_);
152
153 buffer_health_.usage_percent = buffer_health_.total_capacity > 0 ?
154 (static_cast<double>(buffer_health_.used_capacity) /
155 buffer_health_.total_capacity * 100.0) : 0.0;
156
157 buffer_health_.available_capacity =
158 buffer_health_.total_capacity - buffer_health_.used_capacity;
159
160 // Determine health status
161 if (buffer_health_.usage_percent > criteria_.max_buffer_usage_percent ||
162 buffer_health_.available_capacity < criteria_.min_available_buffer_size) {
163 buffer_health_.status = health_status::unhealthy;
164 } else if (buffer_health_.usage_percent > criteria_.max_buffer_usage_percent * 0.8) {
165 buffer_health_.status = health_status::degraded;
166 } else {
167 buffer_health_.status = health_status::healthy;
168 }
169
170 return buffer_health_;
171 }
172
173 queue_health_info check_queue_health_internal() {
174 std::lock_guard<std::mutex> lock(queue_mutex_);
175
176 if (collector_) {
177 auto metrics = collector_->get_queue_metrics();
178 queue_health_.current_size = metrics.first;
179 queue_health_.max_size = metrics.second;
180 }
181
182 queue_health_.usage_percent = queue_health_.max_size > 0 ?
183 (static_cast<double>(queue_health_.current_size) /
184 queue_health_.max_size * 100.0) : 0.0;
185
186 // Determine health status
187 if (queue_health_.usage_percent > criteria_.max_queue_usage_percent ||
188 queue_health_.current_size > criteria_.max_queue_size ||
189 queue_health_.max_wait_time > criteria_.max_queue_wait_time) {
190 queue_health_.status = health_status::unhealthy;
191 } else if (queue_health_.usage_percent > criteria_.max_queue_usage_percent * 0.8 ||
192 queue_health_.dropped_messages > 0) {
193 queue_health_.status = health_status::degraded;
194 } else {
195 queue_health_.status = health_status::healthy;
196 }
197
198 return queue_health_;
199 }

201 void update_writer_stats(const std::string& name,
202 bool success,
203 std::chrono::milliseconds latency) {
204 std::lock_guard<std::mutex> lock(writers_mutex_);
205
206 auto it = writer_health_.find(name);
207 if (it == writer_health_.end()) {
208 return;
209 }
210
211 auto& info = it->second;
212 info.total_writes++;
213 info.last_write_time = std::chrono::system_clock::now();
214
215 if (success) {
216 info.consecutive_failures = 0;
217
218 // Update average latency
219 if (info.avg_write_latency.count() == 0) {
220 info.avg_write_latency = latency;
221 } else {
222 auto total_latency = info.avg_write_latency.count() * (info.total_writes - 1);
223 info.avg_write_latency = std::chrono::milliseconds(
224 (total_latency + latency.count()) / info.total_writes
225 );
226 }
227
228 // Update max latency
229 if (latency > info.max_write_latency) {
230 info.max_write_latency = latency;
231 }
232 } else {
233 info.failed_writes++;
234 info.consecutive_failures++;
235 info.last_failure_time = std::chrono::system_clock::now();
236 }
237
238 // Update health status based on criteria
239 if (info.consecutive_failures >= criteria_.max_consecutive_write_failures ||
240 info.max_write_latency > criteria_.max_write_latency) {
241 info.status = health_status::unhealthy;
242 } else if (info.consecutive_failures > 0 ||
243 info.get_failure_rate() > 5.0) {
244 info.status = health_status::degraded;
245 } else {
246 info.status = health_status::healthy;
247 }
248 }

262 void update_queue_stats(size_t current_size,
263 size_t max_size,
264 bool dropped,
265 std::chrono::milliseconds wait_time) {
266 std::lock_guard<std::mutex> lock(queue_mutex_);
267
268 queue_health_.current_size = current_size;
269 queue_health_.max_size = max_size;
270
271 if (dropped) {
272 queue_health_.dropped_messages++;
273 } else {
274 queue_health_.total_enqueued++;
275 }
276
277 if (wait_time.count() > 0) {
278 queue_health_.total_dequeued++;
279
280 // Update average wait time
281 if (queue_health_.avg_wait_time.count() == 0) {
282 queue_health_.avg_wait_time = wait_time;
283 } else {
284 auto total_wait = queue_health_.avg_wait_time.count() *
285 (queue_health_.total_dequeued - 1);
286 queue_health_.avg_wait_time = std::chrono::milliseconds(
287 (total_wait + wait_time.count()) / queue_health_.total_dequeued
288 );
289 }
290
291 // Update max wait time
292 if (wait_time > queue_health_.max_wait_time) {
293 queue_health_.max_wait_time = wait_time;
294 }
295 }

298 void start_monitoring(std::chrono::seconds interval,
299 std::function<void(const health_check_result&)> callback) {
300 if (monitoring_active_.exchange(true)) {
301 return; // Already monitoring
302 }
303
304 monitoring_thread_ = std::thread([this, interval, callback]() {
305 while (monitoring_active_) {
306 std::unique_lock<std::mutex> lock(monitoring_mutex_);
307 if (monitoring_cv_.wait_for(lock, interval,
308 [this]() { return !monitoring_active_.load(); })) {
309 break; // Monitoring stopped
310 }
311
312 auto result = perform_health_check();
313 if (callback) {
314 callback(result);
315 }
316 }
317 });
318 }
319
320 void stop_monitoring() {
321 if (!monitoring_active_.exchange(false)) {
322 return; // Not monitoring
323 }
324
325 monitoring_cv_.notify_all();
326
327 if (monitoring_thread_.joinable()) {
328 monitoring_thread_.join();
329 }
330 }
331
332 bool is_monitoring() const {
333 return monitoring_active_.load();
334 }

386 std::string format_writer_issues(const writer_health_info& info) {
387 std::ostringstream oss;
388
389 if (info.consecutive_failures > 0) {
390 oss << info.consecutive_failures << " consecutive failures";
391 }
392
393 if (info.max_write_latency > criteria_.max_write_latency) {
394 if (oss.tellp() > 0) oss << ", ";
395 oss << "high latency (" << info.max_write_latency.count() << "ms)";
396 }
397
398 if (info.get_failure_rate() > 5.0) {
399 if (oss.tellp() > 0) oss << ", ";
400 oss << "failure rate " << std::fixed << std::setprecision(1)
401 << info.get_failure_rate() << "%";
402 }
403
404 return oss.str();
405 }

582std::string health_check_utils::health_status_to_string(health_status status) {
583 switch (status) {
584 case health_status::healthy:
585 return "healthy";
586 case health_status::degraded:
587 return "degraded";
588 case health_status::unhealthy:
589 return "unhealthy";
590 case health_status::unknown:
591 return "unknown";
592 default:
593 return "unknown";
594 }
595}

597health_status health_check_utils::aggregate_health_status(const std::vector<health_status>& statuses) {
598 if (statuses.empty()) {
599 return health_status::unknown;
600 }
601
602 // Return the worst status
603 bool has_unhealthy = false;
604 bool has_degraded = false;
605 bool has_unknown = false;
606
607 for (const auto& status : statuses) {
608 if (status == health_status::unhealthy) {
609 has_unhealthy = true;
610 } else if (status == health_status::degraded) {
611 has_degraded = true;
612 } else if (status == health_status::unknown) {
613 has_unknown = true;
614 }
615 }
616
617 if (has_unhealthy) {
618 return health_status::unhealthy;
619 } else if (has_degraded) {
620 return health_status::degraded;
621 } else if (has_unknown) {
622 return health_status::unknown;
623 } else {
624 return health_status::healthy;
625 }
626}

628std::string health_check_utils::format_as_json(const health_check_result& result) {
629 std::ostringstream json;
630 json << "{\n";
631 json << " \"status\": \"" << health_status_to_string(result.get_status()) << "\",\n";
632 json << " \"message\": \"" << result.get_message() << "\",\n";
633 json << " \"issues\": [\n";
634
635 const auto& issues = result.get_issues();
636 for (size_t i = 0; i < issues.size(); ++i) {
637 json << " \"" << issues[i] << "\"";
638 if (i < issues.size() - 1) {
639 json << ",";
640 }
641 json << "\n";
642 }
643
644 json << " ],\n";
645 json << " \"timestamp\": \""
646 << std::chrono::system_clock::to_time_t(result.get_check_time())
647 << "\"\n";
648 json << "}";

650 return json.str();
651}
