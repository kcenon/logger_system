128 static std::unique_ptr<log_formatter_interface> create(
129 format_type type,
130 const format_options& opts = format_options{}
131 ) {
132 switch (type) {
133 case format_type::json:
134 return create_json(opts);
135 case format_type::logfmt:
136 return create_logfmt(opts);
137 case format_type::templated:
138 return create_template(template_formatter::DEFAULT_TEMPLATE, opts);
139 case format_type::compact:
140 return create_compact();
141 case format_type::plain:
142 default:
143 return create_plain(opts);
144 }
