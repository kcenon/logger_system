196 // Format log entry based on output format
197 std::string format_entry(const structured_log_entry& entry) const {
198 switch (format_) {
199 case output_format::json:
200 return format_json(entry);
201 case output_format::logfmt:
202 return format_logfmt(entry);
203 case output_format::plain:
204 return format_plain(entry);
205 default:
206 return format_json(entry);
207 }

210 // Format as JSON (simple implementation without nlohmann/json)
211 std::string format_json(const structured_log_entry& entry) const {
212 std::ostringstream oss;
213 oss << "{";
214
215 // Add timestamp
216 auto time_t = std::chrono::system_clock::to_time_t(entry.timestamp_);
217 oss << "\"" << standard_fields::TIMESTAMP << "\":\"";
218 oss << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%SZ") << "\",";
219
220 // Add standard fields
221 oss << "\"" << standard_fields::LEVEL << "\":\"" << level_to_string(entry.level_) << "\",";
222 oss << "\"" << standard_fields::MESSAGE << "\":\"" << escape_json(entry.message_) << "\",";
223 oss << "\"" << standard_fields::THREAD_ID << "\":\""
224 << std::hash<std::thread::id>{}(std::this_thread::get_id()) << "\"";
225
226 // Add global context
227 for (const auto& [key, value] : global_context_) {
228 oss << ",\"" << key << "\":" << any_to_json_string(value);
229 }
230
231 // Add entry context
232 if (!entry.context_.empty()) {
233 oss << ",\"context\":{";
234 bool first = true;
235 for (const auto& [key, value] : entry.context_) {
236 if (!first) oss << ",";
237 oss << "\"" << key << "\":" << any_to_json_string(value);
238 first = false;
239 }
240 oss << "}";
241 }
242
243 // Add fields
244 for (const auto& [key, value] : entry.fields_) {
245 oss << ",\"" << key << "\":" << any_to_json_string(value);
246 }
247
248 oss << "}";
249 return oss.str();
250 }

252 // Format as logfmt
253 std::string format_logfmt(const structured_log_entry& entry) const {
254 std::ostringstream oss;
255
256 // Add timestamp
257 auto time_t = std::chrono::system_clock::to_time_t(entry.timestamp_);
258 oss << "timestamp=\"" << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%SZ") << "\" ";
259
260 // Add standard fields
261 oss << "level=" << level_to_string(entry.level_) << " ";
262 oss << "message=\"" << escape_string(entry.message_) << "\" ";
263 oss << "thread_id=" << std::hash<std::thread::id>{}(std::this_thread::get_id()) << " ";
264
265 // Add all fields
266 auto add_fields = [&oss, this](const auto& fields) {
267 for (const auto& [key, value] : fields) {
268 oss << key << "=" << format_value_logfmt(value) << " ";
269 }
270 };
271
272 add_fields(global_context_);
273 add_fields(entry.context_);
274 add_fields(entry.fields_);
275
276 return oss.str();
277 }

279 // Format as plain text
280 std::string format_plain(const structured_log_entry& entry) const {
281 std::ostringstream oss;
282
283 // Traditional format with structured data appended
284 oss << entry.message_;
285
286 // Add fields if present
287 if (!entry.fields_.empty()) {
288 oss << " [";
289 bool first = true;
290 for (const auto& [key, value] : entry.fields_) {
291 if (!first) oss << ", ";

293 first = false;
294 }
295 oss << "]";
296 }
297
298 return oss.str();
299 }

301 // Convert std::any to JSON string value
302 std::string any_to_json_string(const std::any& value) const {
303 if (value.type() == typeid(int)) {
304 return std::to_string(std::any_cast<int>(value));
305 } else if (value.type() == typeid(long)) {
306 return std::to_string(std::any_cast<long>(value));
307 } else if (value.type() == typeid(uint64_t)) {
308 return std::to_string(std::any_cast<uint64_t>(value));
309 } else if (value.type() == typeid(double)) {
310 return std::to_string(std::any_cast<double>(value));
311 } else if (value.type() == typeid(float)) {
312 return std::to_string(std::any_cast<float>(value));
313 } else if (value.type() == typeid(bool)) {
314 return std::any_cast<bool>(value) ? "true" : "false";
315 } else if (value.type() == typeid(std::string)) {
316 return "\"" + escape_json(std::any_cast<std::string>(value)) + "\"";
317 } else if (value.type() == typeid(const char*)) {
318 return "\"" + escape_json(std::any_cast<const char*>(value)) + "\"";
319 } else {
320 return "\"unknown\"";
321 }
322 }

324 // Escape string for JSON
325 std::string escape_json(const std::string& str) const {
326 std::string escaped;
327 for (char c : str) {
328 if (c == '"') escaped += "\\\"";
329 else if (c == '\\') escaped += "\\\\";
330 else if (c == '\n') escaped += "\\n";
331 else if (c == '\r') escaped += "\\r";
332 else if (c == '\t') escaped += "\\t";
333 else if (c == '\b') escaped += "\\b";
334 else if (c == '\f') escaped += "\\f";
335 else escaped += c;
336 }
337 return escaped;
338 }

340 // Format value for logfmt
341 std::string format_value_logfmt(const std::any& value) const {
342 if (value.type() == typeid(std::string)) {
343 return "\"" + escape_string(std::any_cast<std::string>(value)) + "\"";
344 } else if (value.type() == typeid(const char*)) {
345 return "\"" + escape_string(std::any_cast<const char*>(value)) + "\"";
346 } else {
347 // For numeric values, use plain format
348 return format_value_plain(value);
349 }
350 }

352 // Format value for plain text
353 std::string format_value_plain(const std::any& value) const {
354 if (value.type() == typeid(int)) {
355 return std::to_string(std::any_cast<int>(value));
356 } else if (value.type() == typeid(long)) {
357 return std::to_string(std::any_cast<long>(value));
358 } else if (value.type() == typeid(uint64_t)) {
359 return std::to_string(std::any_cast<uint64_t>(value));
360 } else if (value.type() == typeid(double)) {
361 return std::to_string(std::any_cast<double>(value));
362 } else if (value.type() == typeid(float)) {
363 return std::to_string(std::any_cast<float>(value));
364 } else if (value.type() == typeid(bool)) {
365 return std::any_cast<bool>(value) ? "true" : "false";
366 } else if (value.type() == typeid(std::string)) {
367 return std::any_cast<std::string>(value);
368 } else if (value.type() == typeid(const char*)) {
369 return std::any_cast<const char*>(value);
370 } else {
371 return "unknown";
372 }
373 }
