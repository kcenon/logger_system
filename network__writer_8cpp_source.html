87result_void network_writer::write(thread_module::log_level level,
88 const std::string& message,
89 const std::string& file,
90 int line,
91 const std::string& function,
92 const std::chrono::system_clock::time_point& timestamp) {
93
94 std::lock_guard<std::mutex> lock(buffer_mutex_);
95
96 // Check buffer size
97 if (buffer_.size() >= buffer_size_) {
98 // Drop oldest message
99 buffer_.pop();
100 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
101 stats_.send_failures++;
102 // Note: We still accept the new message after dropping the oldest
103 }
104
105 buffer_.push({level, message, file, line, function, timestamp});
106 buffer_cv_.notify_one();
107
108 return {}; // Success
109}

111result_void network_writer::flush() {
112 std::unique_lock<std::mutex> lock(buffer_mutex_);
113 auto start = std::chrono::steady_clock::now();
114 auto timeout = std::chrono::seconds(5); // 5 second timeout
115
116 while (!buffer_.empty()) {
117 if (buffer_cv_.wait_for(lock, timeout, [this] { return buffer_.empty() || !running_; })) {
118 if (!buffer_.empty() && !running_) {
119 return make_logger_error(logger_error_code::flush_timeout,
120 "Network writer stopped before flush completed");
121 }
122 } else {
123 return make_logger_error(logger_error_code::flush_timeout,
124 "Network flush timeout");
125 }
126
127 if (std::chrono::steady_clock::now() - start > timeout) {
128 return make_logger_error(logger_error_code::flush_timeout,
129 "Network flush exceeded timeout");
130 }
131 }
132 return {}; // Success
133}

140bool network_writer::connect() {
141 if (connected_) {
142 return true;
143 }
144
145 // Create socket
146 int sock_type = (protocol_ == protocol_type::tcp) ? SOCK_STREAM : SOCK_DGRAM;
147 socket_fd_ = socket(AF_INET, sock_type, 0);
148 if (socket_fd_ < 0) {
149 std::cerr << "Failed to create socket: " << strerror(errno) << std::endl;
150 return false;
151 }
152
153 // Resolve hostname
154 struct hostent* server = gethostbyname(host_.c_str());
155 if (!server) {
156 std::cerr << "Failed to resolve host: " << host_ << std::endl;
157 close(socket_fd_);
158 socket_fd_ = -1;
159 return false;
160 }
161
162 // Setup server address
163 struct sockaddr_in server_addr{};
164 server_addr.sin_family = AF_INET;
165 server_addr.sin_port = htons(port_);
166 std::memcpy(&server_addr.sin_addr.s_addr, server->h_addr, server->h_length);
167
168 // Connect (TCP only)
169 if (protocol_ == protocol_type::tcp) {
170 if (::connect(socket_fd_, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
171 std::cerr << "Failed to connect to " << host_ << ":" << port_
172 << " - " << strerror(errno) << std::endl;
173 close(socket_fd_);
174 socket_fd_ = -1;
175
176 std::lock_guard<std::mutex> lock(stats_mutex_);
177 stats_.connection_failures++;
178 stats_.last_error = std::chrono::system_clock::now();
179 return false;
180 }
181 } else {
182 // For UDP, just save the server address
183 if (::connect(socket_fd_, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
184 std::cerr << "Failed to set UDP destination: " << strerror(errno) << std::endl;
185 close(socket_fd_);
186 socket_fd_ = -1;
187 return false;
188 }
189 }
190
191 connected_ = true;
192
193 std::lock_guard<std::mutex> lock(stats_mutex_);
194 stats_.last_connected = std::chrono::system_clock::now();
195
196 std::cout << "Connected to " << host_ << ":" << port_
197 << " via " << (protocol_ == protocol_type::tcp ? "TCP" : "UDP") << std::endl;
198
199 return true;
200}

210bool network_writer::send_data(const std::string& data) {
211 if (!connected_ || socket_fd_ < 0) {
212 return false;
213 }
214
215#ifdef _WIN32
216 int sent = ::send(socket_fd_, data.c_str(), static_cast<int>(data.length()), 0);
217#else
218 ssize_t sent = ::send(socket_fd_, data.c_str(), data.length(), 0);
219#endif
220 if (sent < 0) {
221 if (protocol_ == protocol_type::tcp) {
222 // TCP connection lost
223 std::cerr << "Send failed: " << strerror(errno) << std::endl;
224 disconnect();
225
226 std::lock_guard<std::mutex> lock(stats_mutex_);
227 stats_.send_failures++;
228 stats_.last_error = std::chrono::system_clock::now();
229 }
230 return false;
231 }
232
233 std::lock_guard<std::mutex> lock(stats_mutex_);
234 stats_.messages_sent++;
235 stats_.bytes_sent += sent;
236
237 return true;
238}

240void network_writer::worker_thread() {
241 while (running_) {
242 std::unique_lock<std::mutex> lock(buffer_mutex_);
243
244 // Wait for logs or shutdown
245 buffer_cv_.wait(lock, [this] {
246 return !buffer_.empty() || !running_;
247 });
248
249 // Process buffered logs
250 while (!buffer_.empty() && running_) {
251 auto log = std::move(buffer_.front());
252 buffer_.pop();
253 lock.unlock();
254
255 // Format and send
256 std::string formatted = format_for_network(log);
257 send_data(formatted);
258
259 lock.lock();
260 }
261 }
262}

275std::string network_writer::format_for_network(const buffered_log& log) {
276 // Format as JSON for network transmission
277 std::ostringstream oss;
278 oss << "{";
279
280 // Timestamp
281 auto time_t = std::chrono::system_clock::to_time_t(log.timestamp);
282 oss << "\"@timestamp\":\"";
283 oss << std::put_time(std::gmtime(&time_t), "%Y-%m-%dT%H:%M:%SZ") << "\",";
284
285 // Level
286 oss << "\"level\":\"" << level_to_string(log.level) << "\",";
287
288 // Message
289 oss << "\"message\":\"" << escape_json(log.message) << "\"";
290
291 // Optional fields
292 if (!log.file.empty()) {
293 oss << ",\"file\":\"" << escape_json(log.file) << "\"";
294 oss << ",\"line\":" << log.line;
295 }
296
297 if (!log.function.empty()) {
298 oss << ",\"function\":\"" << escape_json(log.function) << "\"";
299 }
300
301 // Add hostname
302 char hostname[256];
303 if (gethostname(hostname, sizeof(hostname)) == 0) {
304 oss << ",\"host\":\"" << hostname << "\"";
305 }
306
307 oss << "}\n";
308 return oss.str();
309}

311std::string network_writer::escape_json(const std::string& str) const {
312 std::string escaped;
313 for (char c : str) {
314 if (c == '"') escaped += "\\\"";
315 else if (c == '\\') escaped += "\\\\";
316 else if (c == '\n') escaped += "\\n";
317 else if (c == '\r') escaped += "\\r";
318 else if (c == '\t') escaped += "\\t";
319 else escaped += c;
320 }
321 return escaped;
322}
