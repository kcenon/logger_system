236class performance_tuning_strategy : public config_strategy_interface {
237public:
238 enum class tuning_level {
239 conservative, // Low resource usage
240 balanced, // Default balanced approach
241 aggressive // Maximum performance
242 };
243
244 explicit performance_tuning_strategy(tuning_level level = tuning_level::balanced)
245 : level_(level) {}
246
247 result_void apply(logger_config& config) override {
248 switch (level_) {
249 case tuning_level::conservative:
250 config.buffer_size = 4096;
251 config.batch_size = 50;
252 config.max_queue_size = 1000;
253 config.flush_interval = std::chrono::milliseconds(500);
254 config.writer_thread_count = 1;
255 config.enable_batch_writing = false; // Disable for conservative mode
256 break;
257
258 case tuning_level::balanced:
259 config.buffer_size = 8192;
260 config.batch_size = 100;
261 config.max_queue_size = 10000;
262 config.flush_interval = std::chrono::milliseconds(1000);
263 config.writer_thread_count = 2;
264 config.enable_batch_writing = true; // Enable for balanced mode
265 break;
266
267 case tuning_level::aggressive:
268 config.buffer_size = 65536;
269 config.batch_size = 500;
270 config.max_queue_size = 100000;
271 config.flush_interval = std::chrono::milliseconds(5000);
272 config.writer_thread_count = 4;
273 config.use_lock_free = true;
274 config.enable_compression = true;
275 config.enable_batch_writing = true; // Enable for aggressive mode
276 break;
277 }
278 return result_void{};
279 }
280
281 std::string get_name() const override {
282 switch (level_) {
283 case tuning_level::conservative:
284 return "conservative_tuning";
285 case tuning_level::balanced:
286 return "balanced_tuning";
287 case tuning_level::aggressive:
288 return "aggressive_tuning";
289 }
290 }
291
292 std::string get_description() const override {
293 switch (level_) {
294 case tuning_level::conservative:
295 return "Conservative resource usage for constrained environments";
296 case tuning_level::balanced:
297 return "Balanced performance and resource usage";
298 case tuning_level::aggressive:
299 return "Maximum performance with higher resource consumption";
300 }
301 }
302

308 }
309 return result_void{};
310 }

322class composite_strategy : public config_strategy_interface {
323public:
324 composite_strategy() = default;
325
330 void add_strategy(std::unique_ptr<config_strategy_interface> strategy) {
331 strategies_.push_back(std::move(strategy));
332 // Sort by priority (descending)
333 std::sort(strategies_.begin(), strategies_.end(),
334 [](const auto& a, const auto& b) {
335 return a->get_priority() > b->get_priority();
336 });
337 }
338
339 result_void apply(logger_config& config) override {
340 for (const auto& strategy : strategies_) {
341 if (auto can_apply = strategy->can_apply(config); !can_apply) {
342 continue; // Skip strategies that can't be applied
343 }
344
345 if (auto result = strategy->apply(config); !result) {
346 return result;
347 }
348 }
349 return result_void{};
350 }
351
352 std::string get_name() const override {
353 return "composite";
354 }
355
356 std::string get_description() const override {
357 std::string desc = "Composite of: ";
358 for (size_t i = 0; i < strategies_.size(); ++i) {
359 if (i > 0) desc += ", ";
360 desc += strategies_[i]->get_name();
361 }
362 return desc;
363 }
364
365 result_void can_apply(const logger_config& config) const override {
366 // Composite can apply if at least one strategy can apply
367 for (const auto& strategy : strategies_) {
368 if (auto can_apply = strategy->can_apply(config); can_apply) {
369 return result_void{};
370 }
371 }
372 return make_logger_error(logger_error_code::invalid_configuration,
373 "No strategies in composite can be applied");
374 }
375
376private:
377 std::vector<std::unique_ptr<config_strategy_interface>> strategies_;
378};

392 if (name == "high_performance") {
393 return std::make_unique<template_strategy>(template_strategy::template_type::high_performance);
394 } else if (name == "low_latency") {
395 return std::make_unique<template_strategy>(template_strategy::template_type::low_latency);
396 } else if (name == "debug") {
397 return std::make_unique<template_strategy>(template_strategy::template_type::debug);
398 } else if (name == "production") {
399 return std::make_unique<template_strategy>(template_strategy::template_type::production);
400 } else if (name == "default") {
401 return std::make_unique<template_strategy>(template_strategy::template_type::default_config);
402 }
403 return nullptr;
404 }
405
411 static std::unique_ptr<config_strategy_interface> create_environment(const std::string& env) {
412 if (env == "development" || env == "dev") {
413 return std::make_unique<environment_strategy>(environment_strategy::environment::development);
414 } else if (env == "testing" || env == "test") {
415 return std::make_unique<environment_strategy>(environment_strategy::environment::testing);

417 return std::make_unique<environment_strategy>(environment_strategy::environment::staging);
418 } else if (env == "production" || env == "prod") {
419 return std::make_unique<environment_strategy>(environment_strategy::environment::production);
420 }
421 return nullptr;
422 }
423
429 static std::unique_ptr<config_strategy_interface> create_tuning(const std::string& level) {
430 if (level == "conservative" || level == "low") {
431 return std::make_unique<performance_tuning_strategy>(
432 performance_tuning_strategy::tuning_level::conservative);
433 } else if (level == "aggressive" || level == "high") {
434 return std::make_unique<performance_tuning_strategy>(
435 performance_tuning_strategy::tuning_level::aggressive);
436 }
437 return std::make_unique<performance_tuning_strategy>(
438 performance_tuning_strategy::tuning_level::balanced);
439 }
440
445 static std::unique_ptr<config_strategy_interface> from_environment() {
446 const char* env = std::getenv("LOG_ENV");
447 if (env) {
448 return create_environment(env);
449 }
450
451 const char* level = std::getenv("LOG_LEVEL");
452 if (level && std::string(level) == "DEBUG") {
453 return create_template("debug");
454 }

456 return nullptr;
