76 result<monitoring_data> collect_metrics() const override {
77 if (!enabled_.load()) {
78 return error_code::metrics_not_available;
79 }
80
81 // Try thread_system first
82 if (monitorable_) {
83 try {
84 auto ts_metrics = monitorable_->get_metrics();
85 if (ts_metrics) {
86 // Convert thread_system metrics to our format
87 monitoring_data data;
88
89 // Extract metrics from thread_system monitoring_data
90 // Note: This assumes thread_system has similar metric structure
91 for (const auto& [key, value] : ts_metrics.value().get_values()) {
92 data.add_metric(key, value);
93 }
94
95 // Also add fallback monitor metrics for logger-specific data
96 if (fallback_monitor_) {
97 auto fallback_result = fallback_monitor_->collect_metrics();
98 if (fallback_result) {
99 for (const auto& metric : fallback_result.value().get_metrics()) {
100 data.add_metric(metric);
101 }
102 }
103 }
104
105 return data;
106 }
107 } catch (...) {
108 // Fall through to fallback
109 }
110 }
111
112 // Fallback to basic monitor
113 if (fallback_monitor_) {
114 return fallback_monitor_->collect_metrics();
115 }
116
117 return error_code::metrics_not_available;
118 }

124 result<health_check_result> check_health() const override {
125 health_check_result result;
126
127 // Check thread_system health if available
128 if (monitorable_) {
129 try {
130 auto ts_health = monitorable_->get_health_status();
131 if (ts_health) {
132 // Convert thread_system health to our format
133 auto status = ts_health.value();
134
135 if (!status.is_healthy()) {
136 result.set_status(health_status::degraded);
137 result.add_issue("Thread system reports unhealthy status");
138
139 for (const auto& issue : status.get_issues()) {
140 result.add_issue("Thread system: " + issue);
141 }
142 }
143 }
144 } catch (...) {
145 result.add_issue("Failed to check thread_system health");
146 }
147 }
148
149 // Also check fallback monitor for logger-specific health
150 if (fallback_monitor_) {
151 auto fallback_health = fallback_monitor_->check_health();
152 if (fallback_health) {
153 auto& fb_result = fallback_health.value();
154
155 // Merge health status
156 if (!fb_result.is_healthy()) {
157 if (result.get_status() == health_status::healthy) {
158 result.set_status(fb_result.get_status());
159 } else if (fb_result.get_status() == health_status::unhealthy) {
160 result.set_status(health_status::unhealthy);
161 }
162
163 // Add logger-specific issues
164 for (const auto& issue : fb_result.get_issues()) {
165 result.add_issue(issue);
166 }
167 }
168 }
169 }
170
171 // Set overall message
172 if (result.is_healthy()) {
173 result.set_message("All systems operational");
174 } else {
175 result.set_message("Issues detected in monitoring");
176 }
177
178 return result;
179 }
