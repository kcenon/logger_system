69void logger_crash_safety::initialize(logger_crash_safety_level level,
70 const std::string& emergency_log_path,
71 uint32_t backup_interval_ms) {
72 if (initialized_.exchange(true)) {
73 return;
74 }
75
76 safety_level_ = level;
77 emergency_log_path_ = emergency_log_path;
78 backup_interval_ms_ = backup_interval_ms;
79
80 #ifdef _WIN32
81 emergency_log_fd_ = _open(emergency_log_path_.c_str(),
82 _O_WRONLY | _O_CREAT | _O_APPEND);
83 #else
84 emergency_log_fd_ = open(emergency_log_path_.c_str(),
85 O_WRONLY | O_CREAT | O_APPEND, 0644);
86 #endif
87
88 if (emergency_log_fd_ == -1) {
89 emergency_log_fd_ = STDERR_FILENO;
90 }
91
92 #ifndef _WIN32
93 struct sigaction sa;
94 memset(&sa, 0, sizeof(sa));
95 sa.sa_handler = &logger_crash_safety::signal_handler;
96 sigemptyset(&sa.sa_mask);
97 sa.sa_flags = SA_RESTART;
98
99 for (int sig : {SIGSEGV, SIGABRT, SIGFPE, SIGILL, SIGBUS}) {
100 sigaction(sig, nullptr, &previous_handlers_[sig]);
101 sigaction(sig, &sa, nullptr);
102 }
103 #endif
104
105 if (safety_level_ >= logger_crash_safety_level::standard && auto_backup_enabled_) {
106 start_backup_thread();
107 }
108
109 check_and_recover();
110}

112void logger_crash_safety::register_logger(const std::string& logger_name,
113 std::function<void()> flush_callback,
114 std::function<void(const std::string&)> backup_callback) {
115 std::lock_guard<std::mutex> lock(loggers_mutex_);
116
117 auto it = std::find_if(loggers_.begin(), loggers_.end(),
118 [&logger_name](const logger_entry& entry) {
119 return entry.name == logger_name;
120 });
121
122 if (it != loggers_.end()) {
123 it->flush_callback = flush_callback;
124 it->backup_callback = backup_callback;
125 } else {
126 loggers_.push_back({logger_name, flush_callback, backup_callback});
127 }
128}

139void logger_crash_safety::emergency_log(const std::string& level, const std::string& message) {
140 emergency_log_entry entry;
141 entry.timestamp = std::chrono::system_clock::now();
142 entry.level = level;
143 entry.message = message;
144
145 std::ostringstream oss;
146 oss << std::this_thread::get_id();
147 entry.thread_id = oss.str();
148
149 {
150 std::lock_guard<std::mutex> lock(emergency_log_mutex_);
151 emergency_log_queue_.push(entry);
152
153 while (emergency_log_queue_.size() > max_emergency_entries_) {
154 emergency_log_queue_.pop();
155 }
156 }
157
158 write_emergency_log_entry(entry);
159 total_emergency_logs_++;
160}

254 std::string recovery_marker = emergency_log_path_ + ".recovery";
255
256 if (std::filesystem::exists(recovery_marker)) {
257 emergency_log("WARNING", "Previous crash detected, performing recovery");

274void logger_crash_safety::signal_handler(int signal) {
275 if (g_instance && !g_instance->handling_crash_.exchange(true)) {
276 g_instance->handle_logger_crash(signal);

283 }
284 #endif

289void logger_crash_safety::handle_logger_crash(int signal) {
290 last_emergency_time_ = std::chrono::system_clock::now();
291
292 const char* signal_name = "UNKNOWN";
293 switch (signal) {
294 case SIGSEGV: signal_name = "SIGSEGV"; break;
295 case SIGABRT: signal_name = "SIGABRT"; break;
296 case SIGFPE: signal_name = "SIGFPE"; break;
297 #ifndef _WIN32
298 case SIGILL: signal_name = "SIGILL"; break;
299 case SIGBUS: signal_name = "SIGBUS"; break;
300 #endif
301 }
302
303 signal_safe_emergency_log("FATAL", signal_name);
304
305 if (safety_level_ >= logger_crash_safety_level::standard) {
306 flush_all_loggers();
307 }
308
309 if (safety_level_ == logger_crash_safety_level::paranoid) {
310 backup_all_loggers();
311 }
312
313 std::string recovery_marker = emergency_log_path_ + ".recovery";
314 #ifdef _WIN32
315 int fd = _open(recovery_marker.c_str(), _O_CREAT | _O_WRONLY);
316 #else
317 int fd = open(recovery_marker.c_str(), O_CREAT | O_WRONLY, 0644);
318 #endif
319 if (fd != -1) {
320 close(fd);
321 }
322}

401void logger_crash_safety::signal_safe_write(const char* data, size_t length) {
402 if (emergency_log_fd_ != -1) {
403 size_t written = 0;
404 while (written < length) {
405 ssize_t result = write(emergency_log_fd_, data + written, length - written);
406 if (result > 0) {
407 written += result;
408 } else if (result == -1 && errno != EINTR) {
409 break;
410 }
411 }
412 }

415void logger_crash_safety::signal_safe_emergency_log(const char* level, const char* message) {
416 char buffer[512];
417
418 time_t now = time(nullptr);

421 int len = snprintf(buffer, sizeof(buffer),

437bool log_file_recovery::is_corrupted(const std::string& file_path) {
438 if (!std::filesystem::exists(file_path)) {
439 return false;
440 }
441
442 std::ifstream file(file_path, std::ios::binary | std::ios::ate);
443 if (!file) {
444 return true;
445 }
446
447 std::streamsize size = file.tellg();
448 if (size == 0) {
449 return false;
450 }
451
452 file.seekg(-1, std::ios::end);
453 char last_char;
454 file.get(last_char);
455
456 return last_char != '\n';
457}

459bool log_file_recovery::recover_file(const std::string& file_path, const std::string& recovery_path) {
460 if (!std::filesystem::exists(file_path)) {
461 return false;
462 }
463
464 if (!is_corrupted(file_path)) {
465 return true;
466 }
467
468 std::ifstream source(file_path, std::ios::binary);
469 std::ofstream dest(recovery_path, std::ios::binary);
470
471 if (!source || !dest) {
472 return false;
473 }
474
475 std::string line;
476 while (std::getline(source, line)) {
477 if (!line.empty()) {
478 dest << line << '\n';
479 }
480 }
481
482 return dest.good();
483}

507std::string log_file_recovery::calculate_checksum(const std::string& file_path) {
508 std::ifstream file(file_path, std::ios::binary);
509 if (!file) {
510 return "";
511 }
512
513 uint32_t hash = 0;
514 char buffer[4096];
515
516 while (file.read(buffer, sizeof(buffer)) || file.gcount() > 0) {
517 for (std::streamsize i = 0; i < file.gcount(); ++i) {
518 hash = hash * 31 + static_cast<unsigned char>(buffer[i]);
519 }
520 }
521
522 std::ostringstream oss;
523 oss << std::hex << hash;
524 return oss.str();
525}
